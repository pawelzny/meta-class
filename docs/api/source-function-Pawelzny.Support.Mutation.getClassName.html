14 /**
15  * Returns object's class name without namespace in snake_case
16  * if second parameter is provided will be used instead.
17  *
18  * $name can be integer if you process mixed arrays, where some
19  * keys are numeric and some strings.
20  * When $name is a valid string then will be used as alias name for given object.
21  *
22  * @example
23  * <code>
24  * $objects = [
25  *  $user,                      // instance of User
26  *  $role,                      // instance of Role
27  *  'company' => $workplace     // instance of Workplace
28  * ];
29  * $registry = [];
30  * foreach ($objects as $key => $obj) {
31  *   $registry[] = getClassName($obj, $key);
32  * }
33  *
34  * assertEquals(['user', 'role', 'company'], $registry);
35  * </code>
36  *
37  * @param  mixed          $obj
38  * @param  string|integer $name
39  *
40  * @return string
41  */
42 function getClassName($obj, $name = null)
43 {
44     if (is_string($name) && ! is_numeric($name)) {
45         return $name;
46     }
47     $obj = new \ReflectionClass($obj);
48 
49     return \Pawelzny\Support\Normalization\toSnakeCase($obj->getShortName());
50 }
